while (true) {
  let num1 = parseInt(prompt('Введите первое число (от 0 до 10):'));
  let num2 = parseInt(prompt('Введите второе число (от 0 до 10):'));

  if (num1 >= 0 && num1 <= 10 && num2 >= 0 && num2 <= 10) {
    console.log(Результат умножения: ${num1 * num2});
    break;
  } else {
    console.log('Введенные числа недопустимы. Пожалуйста, введите два числа от 0 до 10.');
  }
}
